CREATE TABLE dog_breeds (
  id INT PRIMARY KEY AUTO_INCREMENT,
  breed_name VARCHAR(255) NOT NULL,
  breed_type ENUM('Purebred', 'Hybrid', 'Mixed') NOT NULL,
  breed_group VARCHAR(255) NOT NULL,
  origin VARCHAR(255) NOT NULL,
  description TEXT NOT NULL,
  image_url VARCHAR(255),
  size VARCHAR(50),
  coat_type VARCHAR(50),
  coat_length VARCHAR(50),
  color VARCHAR(50),
  weight_range VARCHAR(50),
  height_range VARCHAR(50),
  life_expectancy VARCHAR(50),
  health_issue VARCHAR(255),
  temperament_trait VARCHAR(255),
  history TEXT,
  temperament_details TEXT,
  grooming_needs TEXT,
  exercise_needs TEXT,
  training_needs TEXT,
  nutrition_needs TEXT,
  living_conditions TEXT
);
