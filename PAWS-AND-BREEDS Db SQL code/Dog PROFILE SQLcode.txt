CREATE TABLE dog_profiles (
  id INT PRIMARY KEY AUTO_INCREMENT,
  user_id INT,
  dog_name VARCHAR(255),
  breed_id INT NOT NULL,
  age INT,
  sex VARCHAR(10),
  rating TINYINT,
  review_text TEXT,
  image_url VARCHAR(255),
  description TEXT,
  location VARCHAR(255),
  breed_mix VARCHAR(255),
  weight DECIMAL(5,2),
  height DECIMAL(5,2),
  energy_level ENUM('Low', 'Medium', 'High'),
  training_level ENUM('Basic', 'Intermediate', 'Advanced'),
  socialization ENUM('Good', 'Fair', 'Needs improvement'),
  vaccination_status ENUM('Up-to-date', 'Not up-to-date'),
  health_conditions TEXT,
  dietary_restrictions TEXT,
  allergies TEXT,
  temperament TEXT,
  behavioral_issues TEXT,
  fears_phobias TEXT,
  profile_video VARCHAR(255),
  profile_photos VARCHAR(255),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id),
  FOREIGN KEY (breed_id) REFERENCES dog_breeds(id)
);
